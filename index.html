# Example 5: Implementing subcommands in a CLI application

import argparse

def command1(args):
    # Your code for command 1 here
    print("Running command 1")

def command2(args):
    # Your code for command 2 here
    print("Running command 2")

def main():
    parser = argparse.ArgumentParser(description="A CLI application with subcommands")
    subparsers = parser.add_subparsers(title="subcommands", dest="subcommand")

    parser_command1 = subparsers.add_parser("command1", help="Description of command 1")
    parser_command2 = subparsers.add_parser("command2", help="Description of command 2")

    parser_command1.set_defaults(func=command1)
    parser_command2.set_defaults(func=command2)

    args = parser.parse_args()
    args.func(args)

if __name__ == "__main__":
    main()
